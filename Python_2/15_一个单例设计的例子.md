# 一个单例设计的例子

```py
class MusicPlayer(object):
    # 记录第一个被创建的类引用
    instance=None

    def __new__(cls,*args,**kwargs):

        # 1. 判断类属性是否为空对象
        if cls.instance is None:
            # 2.调用父类对象，为第一个对象分配空间
            cls.instance=super().__new__(cls)

        # 3.返回类属性保存的对象引用
        return cls.instance
        
# 创建多个对象

player_1=MusicPlayer()
print(player_1)

player_2=MusicPlayer()
print(player_2)
```
